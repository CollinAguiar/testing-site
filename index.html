def check_strength(pw): length = len(pw) has_upper = any(c.isupper() for c in pw) has_lower = any(c.islower() for c in pw) has_digit = any(c.isdigit() for c in pw) score = length + has_upper*2 + has_lower*2 + has_digit*2 if score > 10: return "Strong" elif score > 6: return "Moderate" else: return "Weak" print("Enter a password in the input below and check strength (open console).")
